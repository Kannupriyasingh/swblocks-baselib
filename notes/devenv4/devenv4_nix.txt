export CI_ENV_ROOT=~/swblocks/ci_env

mkdir -p $CI_ENV_ROOT/scripts/ci

mkdir -p ~/swblocks/dist-devenv4-{centos7|centos8|darwin-15/16/17/18/19|ub18-gcc810-clang800}

mkdir -p ~/swblocks/dist-devenv4-rhel7
OR
mkdir -p ~/swblocks/dist-devenv4-darwin-18
OR
mkdir -p ~/swblocks/dist-devenv4-ub18-gcc810-clang800
OR
mkdir -p ~/swblocks/dist-devenv4-ub18-x86-gcc810

create $CI_ENV_ROOT/scripts/ci/ci-init-env.sh (e.g. gedit $CI_ENV_ROOT/scripts/ci/ci-init-env.sh):

# initialize the important environment roots

export DIST_ROOT_DEPS1=~/swblocks/dist-devenv4-rhel7
export DIST_ROOT_DEPS2=~/swblocks/dist-devenv4-rhel7
export DIST_ROOT_DEPS3=~/swblocks/dist-devenv4-rhel7

export DIST_ROOT_DEPS2=~/swblocks/dist-devenv4-ub18-gcc810-clang800
export DIST_ROOT_DEPS3=~/swblocks/dist-devenv4-ub18-gcc810-clang800
